def perm_sum(li, v):
    if len(li) == 1:
        if li[0] == v:
            return 1
        return 0
    tmp = []
    for i in range(1, len(li)):
        tmp.append(li[i] + li[i - 1])
    if sum(tmp) > v:
        return 0
    return perm_sum(tmp, v)
